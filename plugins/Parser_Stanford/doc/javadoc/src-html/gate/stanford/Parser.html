029     * and Sentence annotations (such as those created by the ANNIE tokenizer and
030     * splitter) already in the inputAS and transforms them into suitable data

033     * parameters.
034     */

105      /**
106       * Parse the current document.  (This is the principal 
107       * method called by a CorpusController.)
108       */
109      public void execute() throws ExecutionException {
110        annotationSet  = document.getAnnotations(annotationSetName);
111        
112        if (debugMode) {
113          System.out.println("Parsing document: " + document.getName());
114        }
115    
116        if (useMapping && (! mappingLoaded) ) {
117          System.err.println("Warning: no mapping loaded!");
118        }
119        
120        
121        if (addConstituentAnnotations || addDependencyFeatures || addDependencyAnnotations) {
122          parseSentences();
123        }
124        else {
125          System.err.println("There is nothing for the parser to do.");
126          System.err.println("Please enable at least one of the \"add...\" options.");
127        }
128      }

131      /**
132       * Initialize the Parser resource.  In particular, load the trained data
133       * file.
134       */
135      public Resource init() throws ResourceInstantiationException {
136        instantiateStanfordParser();
137        if (mappingFile != null) {
138          loadTagMapping(mappingFile);
139        }
140        offsetComparator = new OffsetComparator();
141        super.init();
142        
143        if(tlppClass == null || tlppClass.equals("")) {
144          throw new ResourceInstantiationException(
145                  "TLPP class name must be specified");
146        }
147        try {
148          Class<?> tlppClassObj =
149                  Class.forName(tlppClass, true, Gate.getClassLoader());
150          if(!TreebankLangParserParams.class.isAssignableFrom(tlppClassObj)) {
151            throw new ResourceInstantiationException(tlppClassObj
152                    + " does not implement "
153                    + TreebankLangParserParams.class.getName());
154          }
155          TreebankLangParserParams tlpp =
156                  TreebankLangParserParams.class.cast(tlppClassObj.newInstance());
157          gsf = tlpp.treebankLanguagePack().grammaticalStructureFactory();
158        }
159        catch(UnsupportedOperationException e) {
160          throw new ResourceInstantiationException(e);
161        }
162        catch(ClassNotFoundException e) {
163          throw new ResourceInstantiationException("Class " + tlppClass
164                  + " not found", e);
165        }
166        catch(InstantiationException e) {
167          throw new ResourceInstantiationException("Error creating TLPP object", e);
168        }
169        catch(IllegalAccessException e) {
170          throw new ResourceInstantiationException("Error creating TLPP object", e);
171        }
172        return this;
173      }

186      /**
187       * Find all the Sentence annotations and iterate through them, parsing one
188       * sentence at a time and storing the result in the output AS. (Sentences are
189       * scanned for Tokens. You have to run the ANNIE tokenizer and splitter before
190       * this PR.)
191       */
192      private void parseSentences() { 
193        List<Annotation> sentences = new ArrayList<Annotation>(annotationSet.get(inputSentenceType));
194        java.util.Collections.sort(sentences, offsetComparator);
195        Iterator<Annotation> sentenceIter = sentences.iterator();
196    
197        Tree tree;
198        int debugNbrS, debugS;
199        debugS = 0;
200        debugNbrS = sentences.size();
201    
202        while (sentenceIter.hasNext()) {
203          debugS++;
204          tree = parseOneSentence(sentenceIter.next(), debugS);
205          
206          // Here null is the result from an empty Sentence.
207          if (tree != null) {
208            if (addConstituentAnnotations || addPosTags) {
209              annotatePhraseStructureRecursively(tree, tree);
210            }
211    
212            if (addDependencyFeatures || addDependencyAnnotations) {
213              annotateDependencies(tree);
214            }
215    
216            if (debugMode) {
217              System.out.println("Parsed sentence " + debugS + " of " + debugNbrS);
218            }
219          }
220          
221          else if (debugMode) {
222            System.out.println("Ignored empty sentence " + debugS + " of " + debugNbrS);
223          }
224          
225        }
226      }

229      /**
230       * Generate the special data structure for one sentence and pass the List of
231       * Word to the parser.
232       * 
233       * @param sentence
234       *          the Sentence annotation
235       * @param s
236       *          sentence number of debugging output
237       * @param ofS
238       *          total number of sentences for debugging output
239       * @return  null if the sentence is empty
240       */
241      private Tree parseOneSentence(Annotation sentence, int sentenceNo) {
242        Tree result = null;
243        
244        stanfordSentence = new StanfordSentence(sentence, inputTokenType, annotationSet, reusePosTags);
245    
246        /* Ignore an empty Sentence (sometimes the regex splitter can create one
247         * with no Token annotations in it).
248         */
249        if ( stanfordSentence.isNotEmpty() ) {
250          List<Word> wordList = stanfordSentence.getWordList();
251    
252          if (reusePosTags) {
253            int nbrMissingTags = stanfordSentence.numberOfMissingPosTags();
254            if (nbrMissingTags > 0)  {
255              double percentMissing = Math.ceil(100.0 * ((float) nbrMissingTags) /
256                      ((float) stanfordSentence.numberOfTokens()) );
257              System.err.println("Warning (sentence " + sentenceNo + "): " + (int) percentMissing 
258                      + "% of the Tokens are missing POS tags." );
259            }
260          }
261    
262          stanfordParser.parse(wordList);
263          result = stanfordParser.getBestParse();
264        }
265        
266        return result;
267      }

270      /**
271       * Generate a SyntaxTreeNode Annotation corresponding to this Tree.  Work 
272       * recursively so that the annotations are actually generated from the 
273       * bottom up, in order to build the consists list of annotation IDs.
274       * 
275       * @param tree  the current subtree
276       * @param rootTree  the whole sentence, used to find the span of the current subtree
277       * @return a GATE Annotation of type "SyntaxTreeNode"
278       */
279      protected Annotation annotatePhraseStructureRecursively(Tree tree, Tree rootTree) {
280        Annotation annotation = null;
281        Annotation child;
282        String label   = tree.value();
283    
284        List<Tree> children = tree.getChildrenAsList();
285    
286        if (children.size() == 0) {
287          return null;
288        }
289        /* implied else */
290    
291        /* following line generates ClassCastException
292         *          IntPair span = tree.getSpan();
293         * edu.stanford.nlp.ling.CategoryWordTag
294         * at edu.stanford.nlp.trees.Tree.getSpan(Tree.java:393)
295         * but I think it's a bug in the parser, so I'm hacking 
296         * around it as follows. */
297        int startPos = Trees.leftEdge(tree, rootTree);
298        int endPos   = Trees.rightEdge(tree, rootTree);
299        
300        Long startNode = stanfordSentence.startPos2offset(startPos);
301        Long endNode   = stanfordSentence.endPos2offset(endPos);
302    
303        List<Integer> consists = new ArrayList<Integer>();
304    
305        Iterator<Tree> childIter = children.iterator();
306        while (childIter.hasNext()) {
307          child = annotatePhraseStructureRecursively(childIter.next(), rootTree);
308          if  ( (child != null)  &&
309            (! child.getType().equals(inputTokenType) )) {
310            consists.add(child.getId());
311          }
312        }
313        annotation = annotatePhraseStructureConstituent(startNode, endNode, label, consists, tree.depth());
314    
315        return annotation;
316      }

320      /**
321       * Record one constituent as an annotation.
322       * 
323       * @param startOffset
324       * @param endOffset
325       * @param label
326       * @param consists
327       * @param depth
328       * @return
329       */
330      private Annotation annotatePhraseStructureConstituent(Long startOffset, Long endOffset, String label, 
331        List<Integer> consists, int depth) {
332        Annotation phrAnnotation = null;
333        Integer phrID;
334    
335        try {
336          String cat;
337          if (useMapping && mappingLoaded) {
338            cat  = translateTag(label);
339          }
340          else {
341            cat = label; 
342          }
343          
344          if (addConstituentAnnotations) {
345            String text = document.getContent().getContent(startOffset, endOffset).toString();
346            FeatureMap fm = gate.Factory.newFeatureMap();
347            fm.put(PSG_TAG_FEATURE, cat);
348            fm.put("text", text);
349    
350            /* Ignore empty list features on the token-equivalent annotations. */
351            if (consists.size() > 0) {
352              fm.put("consists", consists);
353            }
354    
355            phrID  = annotationSet.add(startOffset, endOffset, OUTPUT_PHRASE_TYPE, fm);
356            phrAnnotation = annotationSet.get(phrID);
357            recordID(annotationSet, phrID);
358          }
359    
360          if ( addPosTags && (depth == 1) ) {
361            /* Expected to be a singleton set! */
362            AnnotationSet tokenSet = annotationSet.get(inputTokenType, startOffset, endOffset);
363            if (tokenSet.size() == 1) {
364              Annotation token = tokenSet.iterator().next();
365    
366              /* Add POS tag to token.  
367               * (Note: GATE/Hepple uses "(" and ")" for Penn/Stanford's
368               * "-LRB-" and "-RRB-". */
369              String hepCat = StanfordSentence.unescapePosTag(cat);
370              token.getFeatures().put(POS_TAG_FEATURE, hepCat);
371              
372            }
373            else {
374              System.err.println("Found a tokenSet with " + tokenSet.size() + " members!");
375            }
376          }
377        }
378        catch (InvalidOffsetException e) {
379          e.printStackTrace();
380        }
381        
382        return phrAnnotation;
383      }

387      @SuppressWarnings("unchecked")
388      private void annotateDependencies(Tree tree) {
389        GrammaticalStructure gs = gsf.newGrammaticalStructure(tree);
390        Collection<TypedDependency> deps = gs.typedDependencies();
391        String dependencyKind;
392        FeatureMap depFeatures;
393        Integer dependentTokenID, governorTokenID;
394        List<Integer> argList;
395        Long offsetLH0, offsetRH0, offsetLH1, offsetRH1, depLH, depRH;
396        Annotation governorToken, dependentToken;
397    
398        for(TypedDependency d : deps) {
399          if(debugMode) {
400            System.out.println(d);
401          }
402    
403          int governorIndex = ((HasIndex) d.gov().label()).index() - 1;
404          governorToken  = stanfordSentence.startPos2token(governorIndex);
405          
406          int dependentIndex = ((HasIndex) d.dep().label()).index() - 1;
407          dependentToken = stanfordSentence.startPos2token(dependentIndex);
408    
409          dependencyKind = d.reln().toString();
410          governorTokenID = governorToken.getId();
411          dependentTokenID = dependentToken.getId();
412    
413          if (addDependencyFeatures) {
414            List<DependencyRelation> depsForTok =
415              (List<DependencyRelation>) governorToken.getFeatures().get(dependenciesFeature);
416    
417            if(depsForTok == null) {
418              depsForTok = new ArrayList<DependencyRelation>();
419              governorToken.getFeatures().put(dependenciesFeature, depsForTok);
420            }
421    
422            depsForTok.add(new DependencyRelation(dependencyKind, dependentTokenID));
423          }
424          
425          
426          if (addDependencyAnnotations) {
427            depFeatures = gate.Factory.newFeatureMap();
428            argList = new ArrayList<Integer>();
429            argList.add(governorTokenID);
430            argList.add(dependentTokenID);
431            depFeatures.put(DEP_ARG_FEATURE, argList);
432            depFeatures.put(DEP_LABEL_FEATURE, dependencyKind);
433          
434            offsetLH0 = governorToken.getStartNode().getOffset();
435            offsetRH0 = governorToken.getEndNode().getOffset();
436            offsetLH1 = dependentToken.getStartNode().getOffset();
437            offsetRH1 = dependentToken.getEndNode().getOffset();
438          
439            depLH = Math.min(offsetLH0, offsetLH1);
440            depRH = Math.max(offsetRH0, offsetRH1);
441          
442            try {
443              annotationSet.add(depLH, depRH, DEP_ANNOTATION_TYPE, depFeatures);
444            }
445            catch(InvalidOffsetException e) {
446              e.printStackTrace();
447            }
448          }
449        }
450      }

466      private void loadTagMapping(File mappingFile)  { 
467        tagMap = new HashMap<String, String>();
468        mappingLoaded = false;
469    
470        try {
471          if (mappingFile.exists() && mappingFile.canRead()) {
472    
473            BufferedReader br = new BufferedReader(new FileReader(mappingFile));
474            String line = "";
475    
476            // read until it reaches to an end of the file
477            while((line = br.readLine()) != null) {
478              // two columns delimited by whitespace 
479              String [] data = line.split("\\s+", 2);
480    
481              // are there key and value available
482              if(data == null || data.length < 2) {
483                continue;
484              } else {
485                // and add it to the map
486                tagMap.put(data[0].trim(), data[1].trim());
487              }
488            }
489    
490            br.close();
491          }
492    
493          else {
494            System.err.println("Can't find or read mapping file " 
495              + mappingFile.getPath() + " so no mappings will be used.");
496          }
497        } 
498        catch(Exception e) {
499          System.err.println("Exception trying to load mapping file "
500            + mappingFile.getPath());
501          e.printStackTrace();
502        }
503    
504        int nbrMapped = tagMap.size();
505        System.out.println("Loaded " + nbrMapped + " mappings from file " + mappingFile);
506        mappingLoaded = (nbrMapped > 0);
507      }

510      /**
511       * This method stores the annotation ID as a value of feature "ID" on the
512       * relevant annotation. (Mainly to make the ID visible in the GUI for
513       * debugging.)
514       * 
515       * @param annSet
516       * @param annotationID
517       */
518      private void recordID(AnnotationSet annSet, Integer annotationID) {
519        annSet.get(annotationID).getFeatures().put("ID", annotationID);
520      }

523      /**
524       * Translate the tag in the map, or leave it the same if there is no
525       * translation.
526       * 
527       * @param stanfordTag
528       * @return
529       */
530      private String translateTag(String stanfordTag) {
531        String translatedTag = stanfordTag;
532    
533        if (tagMap.containsKey(stanfordTag)) {
534          translatedTag = tagMap.get(stanfordTag);
535        }
536        
537        return translatedTag;
538      }
